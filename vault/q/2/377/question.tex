
\question Consider the following relational schema,\\
\begin{tabular}{ll}
  Table           &(field1, field2, ...)\\
  \hline
  \texttt{GradStudent}     &\texttt{(gsid, firstname, lastname, birthdate, address)} \\
  \texttt{WorksOn}         &\texttt{(gsid, pid, hours)} \\
  \texttt{ResearchProject} &\texttt{(pid, pname, lab)} \\
  \hline \\
\end{tabular}
where \texttt{gsid} (Grad Student Id) uniquely identifies a graduate student and
\texttt{pid} (Project Id) uniquely identifies a research project. Now write the 
algebraic expressions for the following queries

\begin{parts}
  \part[3] Find the first and last names of students who work on 
  projects in the \textbf{Graphics} Lab.

\begin{solution}[\halfpage]
    \[
      \pi_{\texttt{firstname, lastname}}
      \left(
        \underbrace{
          \pi_{\texttt{gsid}}\left(
            \overbrace{ 
              \pi_{\texttt{pid}}\left(
                \sigma_{\texttt{lab='Graphics'}}\texttt{ResearchProject}
              \right)
            }
            ^{\text{IDs of projects in the Graphics Lab}}
            \bowtie\texttt{WorksOn}
          \right)
        }
        _{\text{IDs of students working on projects in the Graphics Lab}}
        \bowtie\texttt{GradStudent}
      \right)
    \]
  \end{solution}

  \part[3] Find the \texttt{gsid} of graduate students who work either in
  the \textbf{Graphics} lab or in the \textbf{Databases} lab 

\begin{solution}[\halfpage]
    \[
      \pi_{\texttt{gsid}}\left(
        \underbrace{ 
          \pi_{\texttt{pid}}\left(
            \sigma_{\texttt{lab='Graphics' $\vee$ lab='Databases'}}\texttt{ResearchProject}
          \right)}
        _{\text{IDs of projects in either of the two labs}}
        \bowtie\texttt{WorksOn}
      \right)
    \]
  \end{solution}

\end{parts}
