\ADD\vbthree{1}\ttl

\question[3] Find the middle terms in the expansion of 
$\left(\vbone x - \dfrac{x^2}{\vbtwo}\right)^{\vbthree}$.


\watchout

\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
\fi 

\begin{solution}[\halfpage]
  Since the expansion has $\ttl$ terms, the middle terms are the \\
  $\left(\dfrac{\ttl}{2}\right)=\tone_{th}$ and 
  $\left(\dfrac{\ttl}{2}+1\right)=\ttwo_{th}$ terms.
  
  \begin{align}
    T_{\tone} &= T_{\o +1} \\
              &= ^{\vbthree}C_\o (\vbone x)^\ocomp \left(\dfrac{x^2}{\vbtwo}\right)^\o (-1)^\o \\
              &= \vbnine\dfrac{\vbfour}{\vbfive} x^{\vbsix} \\
    T_{\ttwo} &= T_{\w +1} \\
              &= ^{\vbthree}C_\w (\vbone x)^\wcomp \left(\dfrac{x^2}{\vbtwo}\right)^\w (-1)^\w \\
              &= \vbten\dfrac{\vbfour}{\vbseven} x^{\vbeight}
  \end{align}

\end{solution}
