
%same suit
\renewcommand{\va}{All 4 cards are of the same suit}
\renewcommand{\ve}
  {First pick one of the four suits ($4$ ways). Then, pick $4$ from the $13$ 
   cards in that suit. Combining these we get $4\times ^{13}C_4=2860$ ways}
\renewcommand{\vi}{2860}
   
%same number
\renewcommand{\vb}{All 4 cards have the same number}
\renewcommand{\vf}
  {The first card can be picked in $13$ ways, since there are $13$ unique 
   \textit{numbers} including face cards. The next three cards are now fixed
   as that same number, but from the rest of the suits. Therefore final answer
   is $13$ ways}
\renewcommand{\vj}{13}

%face cards
\renewcommand{\vc}{All 4 cards are face cards (Jack, Queen, King)}
\renewcommand{\vg}
  {There are $3$ face cards in each suit and $4$ suits which makes for a total
   of $4\times 3=12$ face cards. Number of ways to choose $4$ from $12$ is
   $^{12}C_4=495$}
\renewcommand{\vk}{495}

%succession
\renewcommand{\vd}{All 4 cards are in succession and of the same suit}
\renewcommand{\vh}
  {The first card can be picked in $52$ ways since there are $52$ unique cards. 
   Thereafter the next three cards, starting at the first card we picked, are 
   fixed as successive cards of the same suit. Therefore, number of ways is 
   simply $52$}
\renewcommand{\vl}{52}
