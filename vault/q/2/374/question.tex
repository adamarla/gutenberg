\question One and only one of the following four people $John$, $Boris$, 
$Ivan$ or $Stephan$ will win the upcoming $Wimbledon$ tennis championship 
next month. You are told that the chances of $\va$ winning are the 
same as $\vb$. $\ve$ chances are $\vg$ that of $\va$. 
$\vf$ chances are $\vh$ times that of $\va$.


\watchout

\begin{parts}
  \part[2] Determine the probability of each of those players individually
  winning the $Wimbledon$ tennis championship.

\begin{solution}[\mcq]
    If the chances of $\va$ winning is $x$, then the probabilities 
    for each of them winning are,
    \begin{align}
      P_{\va}   &= \dfrac{x}{x+x+2x+4x} = \dfrac{1}{\b} \\
      P_{\vb}   &= \dfrac{x}{x+x+2x+4x} = \dfrac{1}{\b} \\
      P_{\vc} &= \dfrac{2x}{x+x+2x+4x} = \dfrac{\vi}{\b} \\
      P_{\vd}  &= \dfrac{4x}{x+x+2x+4x} = \dfrac{\vj}{\b}
    \end{align}
  \end{solution}

  \part[1] Find the probability that $\va$ will lose the championship.
\begin{solution}[\mcq]
    Probability that $\va$ will lose is complement of his probability
    to win. Therefore,
    \begin{align}
      P &= 1 - P_{\va} \\
        &= \dfrac{\c}{\b}
    \end{align}    
  \end{solution}

  \part[1] Find the probability of $\vc$ or $\vd$ winning the 
  championship.
\begin{solution}[\mcq]
    Union of probabilities of $\vc$ or $\vd$ winning,
    \begin{align}
      P &= P_{\vc} + P_{\vd} \\
        &= \dfrac{\vi}{\b} + \dfrac{\vj}{\b} \\
        &= \dfrac{\a}{\b}
    \end{align}
  \end{solution}

\end{parts}
