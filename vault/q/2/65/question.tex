

\question[4]  Prove that
	\begin{align}
		\dfrac{\sin 2\theta -\sin 3\theta +\sin 4\theta}
			  {\cos 2\theta -\cos 3\theta +\cos 4\theta} = \tan 3\theta \nonumber
	\end{align}

\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
  \marginnote[1cm] {Key insight is that since the angle on the Right Hand Side is $3\theta$, try and convert terms on the Left Hand Side to $3 \theta$ as well.}
\fi 

\begin{solution}[\halfpage]
	Start with the Left Hand Side (LHS),
	\begin{align}
		\text{LHS} &= \dfrac{\sin 2\theta -\sin 3\theta +\sin 4\theta}
			    		    {\cos 2\theta -\cos 3\theta +\cos 4\theta} \\
			       &= \dfrac{\sin \left(3\theta - \theta\right) + 
			       			 \sin \left(3\theta + \theta\right) - \sin 3\theta}
			       			{\cos \left(3\theta - \theta\right) + 
			       			 \cos \left(3\theta + \theta\right) - \cos 3\theta} \\
			       &= \dfrac{2\sin 3\theta\cos \theta -\sin 3\theta}
			       			{2\cos 3\theta\cos \theta - \cos 3\theta} \\
			       &= \dfrac{\sin 3\theta}{\cos 3\theta} = \tan 3\theta
	\end{align}
\end{solution}
