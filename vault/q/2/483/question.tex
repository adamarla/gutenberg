\question[3] If $y=\dfrac{e^x - e^{-x}}{e^x + e^{-x}}$, prove that 
$\dfrac{dy}{dx}=\dfrac{4}{\left(e^x + e^{-x}\right)^2}$.

\begin{solution}[\mcq]
  Use \textit{Quotient Rule},
  \begin{align}
      \dfrac{dy}{dx}&=\dfrac{(e^x + e^{-x})(e^x + e^{-x})-
                             (e^x - e^{-x})(e^x - e^{-x})}
                            {(e^x + e^{-x})^2} \\
                    &={e^{2x}+1+1+e^{-2x}-e^{2x}+1+1-e^{-2x}}
                            {(e^x + e^{-x})^2} \\
                    &=\dfrac{4}{(e^x + e^{-x})^2}
  \end{align}
\end{solution}
