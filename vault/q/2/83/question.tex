

\question[3]  Vidya received \texteuro2,000 from her grandparents when she graduated from
school and entered college. She placed the money in an account that earns
her 12\% interest(annual) compounded semi-annually. How much will Vidya have when she 
graduates in 4 years time?


\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
  \marginnote[1cm]{
    \begin{align}
      A &= \text{\texteuro 2,000}\cdot (1.06)^8 \\
      \Rightarrow \log A &= \log 2,000 + 8\cdot\log 1.06 \\
                         &= 3.50347 \\
      \Rightarrow A &= \text{\texteuro 3,187.7}
    \end{align}
  }
\fi 

\begin{solution}[\halfpage]
  If $A$ be the amount Vidya has after $4$ years, then
	\begin{align}
			 \text{A} &= \text{\texteuro 2,000}\cdot\left(1 + \dfrac{0.12}{2}\right)^{2\times 4} \\
			          &= \text{\texteuro 2,000}\cdot 1.06^8 \\
                &= \text{\texteuro 3,187.7}
  \end{align}
\end{solution}
