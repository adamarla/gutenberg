\question[2] If $\sqrt{\va} = \vc$ and $\sqrt{\vb}=\vd$, 
then find the value of 
\begin{align}
  \dfrac{\ve}{\va\sqrt{\va}-\vb\sqrt{\vb}}+
    \dfrac{\vf}{\va\sqrt{\va}+\vb\sqrt{\vb}} \nonumber
\end{align}

\watchout

\begin{solution}[\mcq]
  \begin{align}
      &\dfrac{\ve}{\va\sqrt{\va}-\vb\sqrt{\vb}}+
        \dfrac{\vf}{\va\sqrt{\va}+\vb\sqrt{\vb}} \\
    = &\dfrac{\ve (\va\sqrt{\va}+\vb\sqrt{\vb}) +
         \vf (\va\sqrt{\va}-\vb\sqrt{\vb})}
         {(\va\sqrt{\va})^2 - (\vb\sqrt{\vb})^2} \\
    = &\dfrac{\vg\sqrt{\va} + \vh\sqrt{\vb}}{\vi} \\
    = &\vj
  \end{align}
\end{solution}

\ifprintanswers\begin{codex}$\vj$\end{codex}\fi
