\question If the letters of the word $\va$ are arranged at 
random. Find the probability that,

\begin{parts}
  \part All $\vb$s come consecutively in a word.

\begin{solution}[\mcq]
    Let the number of arrangements possible with all
    4 $\vb$s together be $N_{\vc}$.
    \begin{align}
      N_{\vc} &= \dfrac{\a!}{\vk}
    \end{align}
    Total number of arrangements of the word,
    \begin{align}
      N_{total} &= \dfrac{\vi!}{\vj}
    \end{align}
    Probability that 4 $\vb$s come together,
    \begin{align}
      P_{\vc} &= \dfrac{N_{\vc}}{N_{total}} \\
              &= \dfrac{\a!}{\vk}\times\dfrac{\vj}{\vi!} 
               = \vl
    \end{align}
  \end{solution}

  \part All the $\vg$ are together \textit{e.g. \vh}

  \begin{solution}
    Let the number of arrangements possible with the $\vg$
    together be $N_{\vg}$. We will calculate this by finding
    the number of arrangements of $\vg$ in of themselves and
    multiply this by the number of arrangements of the rest of the
    letters but with the $\vg$ treated as a single letter,
    \begin{align}      
      N_{\vg} &= \underbrace{\vt}_\text{\vg only}\times
                      \overbrace{\dfrac{\vv!}{\vu}}^\text{rest of the letters}
    \end{align}
    Probability that $\vg$ come together,
    \begin{align}
      P_{\vg} &= \dfrac{N_{\vg}}{N_{total}} \\
                   &= \vt\times\dfrac{\a!}{\vu}
                      \times\dfrac{\vj}{\vi!}
                    = \vm
    \end{align}
  \end{solution}

  \part Not all $\vf$ come together. \textit{(includes cases
  where two $\vf$ are together but the third is not e.g.\vn)}

  \begin{solution}
    To find the probability that all $\vf$ are not together,
    find the probability that all $\vf$ are together and take
    it's complement.\\
    Let the number of arrangements with all $\vf$ together be
    $N_{\vd}$.
    \begin{align}
      N_{\vd} &= \dfrac{\b}{\vo}
    \end{align}
    Probability that all $\vf$ are together,
    \begin{align}
      P_{\vd} &= \dfrac{N_{\vd}}{N_{total}} \\
                  &= \dfrac{\b}{\vo}\times\dfrac{\vj}{\vi!}
                   = \vp
    \end{align}
    Therefore, probability that not all $\vf$ are together,
    \begin{align}
      P'_{\vd} &= 1 - P_{\vd} \\
                   &= \vq
    \end{align}
  \end{solution}

  \part No two $\vf$ come together i.e. no $\vf$ in the word 
  has another $\vf$ adjacent to it.\textit{e.g. \va}

  \begin{solution}
    Let probability that exactly two $\vf$ occur together in a
    word be $P_{\ve}$ and $none$ of the $\vf$ occur together 
    in a word be $P_\vf$. Using this notation and $P_{\vd}$ 
    from c) we can declare that,
    \begin{align}
      P_{\vd} + P_{\ve} + P_{\vf} = 1
    \end{align}
    We already know $P_{\vd}$ from c). Next let us find
    $P_{\ve}$. The two $\vf$ may occur on the edge 
    (beginning or end) of the word, or somewhere in the middle. 
    Each of these need to be counted separately ($N_{edge}$, 
    $N_{middle}$).
    \begin{align}
      N_{\ve} &= N_{edge} + N_{middle} \\
                  &= \dfrac{\a\times 2\times \a!}{\vo} +
                     \dfrac{\a\times \c\times \a!}{\vo} \\
                  &= \dfrac{\a\times \b!}{\vo}
    \end{align}
    Therefore, probability that exactly two $\vf$ are together,
    \begin{align}
      P_{\ve} &= \dfrac{N_{\ve}}{N_{total}} \\
                  &= \dfrac{\a\times \b!}{\vo}
                     \times\dfrac{\vj}{\vi!}
                   = \vr
    \end{align}
    Therefore, probability that each of the three $\vf$ are by
    themselves,
    \begin{align}
      P_{\vf} &= 1 - P_{\vd} - P{\ve} \\
                 &= 1 - \vp - \vr \\
                 &= \vs
    \end{align}  
  \end{solution}

\end{parts}
