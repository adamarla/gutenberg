\begin{solution}[\mcq]
\ifnumequal{\value{rolldice}}{0}{
The number of letters in the word "ASSASSINATION" : $13$\\
Number of vowels : 6\\
Thus,\\
\text{The Probability of getting a vowel is} $\dfrac{6}{13}$\\
}\\
\ifnumequal{\value{rolldice}}{1}{

}\\
\ifnumequal{\value{rolldice}}{2}{

}\\
\end{solution}
