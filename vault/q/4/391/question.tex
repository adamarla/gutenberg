\question For the equation $\va x^2 - \vb x + \vc = 0$,

\watchout

\begin{parts}
  \part[3] Determine whether the equation has real roots.

  \begin{solution}[\mcq]
    For the equation to have real roots, 
    \[ D = \text{discriminant} = b^2-4ac \geq 0 \]
    And in this case 
    \[ D = \vb^2-4\times\va\times\vc = \vd \geq 0 \implies\text{ real roots} \]
  \end{solution}

  \part[4] If the equation has real roots, then find the roots by first expressing 
  the equation in form 
  \[ (x-a)^2-b^2 = 0 \]

  \begin{solution}[\halfpage]
    \begin{align}
      &\va x^2-\vb x + \vc = 0 \implies\va\times(\va x^2-\vb x + \vc) = 0 \\
      &\implies (\va x)^2 - \left( 2\times\dfrac\vb{2}\right)\cdot\va x + \vg 
      + \underbrace{\left[\left(\dfrac\vb{2}\right)^2-\left( \dfrac\vb{2}\right)^2 \right]}_{\text{effectively 0}} = 0 \\
      &\implies \underbrace{\left[ (\va x)^2 - 2\cdot\dfrac\vb{2}\cdot\va x + \left(\dfrac\vb{2}\right)^2 \right]}_{(a-b)^2}
      + \left[ \vg - \left(\dfrac\vb{2} \right)^2\right] = 0 \\
      &\implies \underbrace{\left( \va x - \dfrac\vb{2} \right)^2 - \dfrac\vr\vs}_{a^2-b^2} = 0 \\
      &\implies\left( \va x -\dfrac\vb{2}+\dfrac\vj\vk \right)\cdot
      \left( \va x -\dfrac\vb{2} - \dfrac\vj\vk \right) = 0 \\
      &\implies x = \WRITEFRAC[false]\w\x,\WRITEFRAC[false]\y\z
    \end{align}
  \end{solution}
\end{parts}
