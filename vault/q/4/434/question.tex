\begin{parts}
  \part[3] "$\vc$" is included in each word

\begin{solution}[\mcq]
  When one letter is fixed as $\vc$. We choose $\vd$ letters from the remaining letters of $\vb$ other than "$\vc$".\\
  And we arrange these letters in $\va !$ ways.\\
  Total number of words when "$\vc$" is included in each word 
  \begin{align}
  \implies ^{7}C_{\va} \times {\va}!  = \vf
  \end{align}
  \end{solution}

  \part[3] "$\vc$" is not included in each word

\begin{solution}[\mcq]
  We choose $\va$ letters from the remaining letters of $\vb$ other than "$\vc$".\\
  And we arrange these letters in $\va !$ ways.\\
  Total number of words when "$\vc$" is not included in any word 
  \begin{align}
  \implies ^{\ve}C_{\va} \times {\va}! = \vg
  \end{align}
  \end{solution}


\end{parts}
