\question Solve the inequality : \\
  $\quad  \dfrac{\vbone x - 3}{4} + \vbtwo  \geq \vbthree$


\watchout

\begin{solution}
\begin{align}
\dfrac{\vbone x}{4} - \dfrac{3}{4} + \vbtwo &\geq \vbthree \\
 \dfrac{4}{3}-\dfrac{\vbone}{4} \cdot x &\leq \vbtwo - 3 - \dfrac{3}{4} \\
\vbfour x &\leq \vbfive \\
 \therefore \quad \quad  x &\leq \vbsix 
\end{align}
\end{solution}
