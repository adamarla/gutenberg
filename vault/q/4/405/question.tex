\question[3] Solve ${\vbone} x - \vbtwo \geq {\vbthree} x - \vbfour$


\watchout

\begin{solution}[\mcq]
\begin{align}
\vbone x - \vbtwo \geq \vbthree x - \vbfour \\
\left( \vbone - \vbthree \right) x \geq \left( \vbtwo - \vbfour \right)\\ 
\vbfive x \geq \vbsix \\
\text{Thus,} \quad \quad
x \geq \vbseven
\end{align}
\end{solution}
