\question There are $\vbone$ doctors in a surgical department. In how many ways they can be arranged to form the following teams

\begin{parts}
  \part  A surgeon and an assistant 

  \begin{solution}
  First we select a surgeon from the $\vbone$ available doctors in $\encr{\vbone}{1}$ ways.\\
  From among the rest we select an assistant in $\encr{\k}{1}$ ways.\\
  Thus total number of arrangements are $\encr{\vbone}{1} \cdot \encr{\k}{1} = \l$ ways. 
  \end{solution}

  \part A surgeon and four assistants 

  \begin{solution}
  First we select a surgeon from the $\vbone$ available doctors in $\encr{\vbone}{1}$ ways.\\
  From among the rest we select four assistants $\encr{\k}{4}$ ways.\\
  Thus total number of arrangements are \\ $\encr{\vbone}{1} \cdot \encr{\k}{4} = \vbone \cdot \encr{\k}{4}$ ways.
  \end{solution}
\end{parts}
