

\question[1] Prove that $\sqrt{\dfrac{1-\sin A}{1+\sin A}} = \sec A - \tan A$

\begin{solution}[\mcq]
   \begin{align}
      \sqrt{\dfrac{1-\sin A}{1+\sin A}} &= 
      \sqrt{\dfrac{(1-\sin A)\cdot(1-\sin A)}{\underbrace{(1+\sin A)\cdot(1-\sin A)}_{\cos^2 A}}} \\
      &= \dfrac{1-\sin A}{\cos A} \\
      &= \sec A - \tan A
   \end{align}
\end{solution}
