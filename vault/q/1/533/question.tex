\question[4] For what value of $N$ would the following be true? 
 \[ \log_{\va}\a + \log_{\vc} N = \log_{\vd}\d \] 

\insertQR{}

\watchout

\begin{solution}[\halfpage]
  \begin{align}
    \va^\vb &= \a \implies \log_{\va}\a = \vb \\
    \vd^\ve &= \d \implies \log_{\vd}\d = \ve \\
    \therefore \log_{\va}\a + \log_{\vc} N &= \log_{\vd}\d \implies \vb + \log_{\vc} N = \ve \\
    \implies \log_{\vc} N &= \ve-\vb = \z \\
    \implies N &= \vc^{\z} = \WRITEFRAC\vx\vy
  \end{align}
\end{solution}
