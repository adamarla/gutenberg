\question[1] Find $N$ if $\dfrac{\enpr{N}\va}{\enpr{N-1}\va} = \dfrac{\vb}{\vc}$

\watchout

\ifprintanswers
\fi 

\begin{solution}[\mcq]
	\begin{align}
		\dfrac{\enpr{N}\va}{\enpr{N-1}\va} &= \fnpr{N}\va\times\overbrace{\dfrac{(N-1-\va)!}{(N-1)!}}^{(N-\a)!} \\
		&= \dfrac{N}{N-\va} \\
		\text{ And }\therefore \dfrac{N}{N-\va} &= \dfrac{\vb}{\vc} \implies N = \ans
	\end{align}
\end{solution}
