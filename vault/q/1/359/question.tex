\question[4] Express $\va + \vb + \vc$ in the form $\frac{p}{q}$, where $q\neq 0$.

\watchout

\begin{solution}[\halfpage]
	\begin{align}
		S = \va + \vb + \vc &= \va \nonumber\\
								& + \vd \nonumber\\
								& + \ve \\
								&= \vf = \vj \\
		\implies 10S &=  \vg \\
		\text{and } 100S &= \vh \\
		\therefore 100S - 10S = \vi \implies S = \WRITEFRAC[false]\vi{90}
	\end{align}
	
\end{solution}
