

\question[2]  If two chords $AB$ and $CD$ of a circle intersect at a point $X$ inside
the circle, then which of the following are true? Select all that apply


\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
\fi 
\checkboxchar{$\Box$}
\begin{checkboxes}
  \CorrectChoice {$AX\cdot BX = CX\cdot DX$}
  \choice {$AX+BX=CX+DX$}
  \CorrectChoice {$\dfrac{AX}{CX}=\dfrac{DX}{BX}$}
  \choice {None of the above}
\end{checkboxes}
