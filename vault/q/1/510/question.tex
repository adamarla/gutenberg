\ADD\va\vd\vb
\ADD\vb\vd\vc

\MULTIPLY\va\ve\vi
\MULTIPLY\vb\ve\vj
\MULTIPLY\vc\ve\vk

\ADD\va\vf\vl
\ADD\vb\vf\vm
\ADD\vc\vf\vn

\SUBTRACT\va\vh\vo
\SUBTRACT\vb\vh\vp
\SUBTRACT\vc\vh\vq


\question If $X = \lbrace \va,\vb,\vc\rbrace$, then write the following sets with all their members 

\watchout

\ifprintanswers
\fi 

\begin{parts}
  \part[1] $A = \lbrace x : x=\ve n, n\in X \rbrace$ 

\begin{solution}[\mcq]
    $A = \lbrace\vi,\vj,\vk\rbrace$
  \end{solution}

  \part[1] $B = \lbrace x : x=n + \vf, n \in X \rbrace$ 

\begin{solution}[\mcq]
    $B = \lbrace\vl,\vm,\vn\rbrace$
  \end{solution}

  \part[1] $C=\lbrace x : x=\frac{n}{\vg}, n \in X \rbrace$

\begin{solution}[\mcq]
    $C=\lbrace\WRITEFRAC\va\vg,\WRITEFRAC\vb\vg,\WRITEFRAC\vc\vg\rbrace$
  \end{solution}

  \part[1] $D = \lbrace x: x=n-\vh, n \in X \rbrace$

\begin{solution}[\mcq]
    $D=\lbrace\vo,\vp,\vq\rbrace$
  \end{solution}
\end{parts}
