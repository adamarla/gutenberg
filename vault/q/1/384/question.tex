\question[3] Find the following limit 
\[ \lim_{n\to\infty}\left( \frac{1^\va}{n^\vb} + \frac{2^\va}{n^\vb} + \ldots
+ \frac{(n-1)^\va}{n^\vb}\right)\]

\watchout[-40pt]

\begin{solution}[\halfpage]
  The above expression can be re-written as 
  \begin{align}
    &\lim_{n\to\infty}\left( \frac{1^\va}{n^\vb} + \frac{2^\va}{n^\vb} + \ldots
+ \frac{(n-1)^\va}{n^\vb}\right) \\ 
  &= \lim_{n\to\infty}\dfrac{1}{n^\vb}\cdot\sum_{k=1}^{n-1}k^\va \\
  &= \lim_{n\to\infty}\dfrac{1}{n^\vb}\cdot\vc \\
  &= \lim_{n\to\infty}\dfrac{1}{n^\vb}\cdot\vd = \ve
  \end{align}
\end{solution}
