\SQUARE\va\vc
\SQUARE\vb\vd
\MULTIPLY\vb{2}\ve

\EXPR[0]\a{\vc - \vd}
\EXPR[0]\b{4*\vc - \vd}
\EXPR[0]\c{9*\vc - \vd}

\EXPR[0]\m{\va - \vb}
\EXPR[0]\n{2*\va - \vb}
\EXPR[0]\o{3*\va - \vb}
\EXPR[0]\p{4*\va - \vb}

\question[3] Find an expression for the sum of the first $n$ terms of the series 
\[ \dfrac\ve\a + \dfrac\ve\b + \dfrac\ve\c + \ldots + \dfrac\ve{\vc n^2 - \vd } + \ldots \]

\begin{solution}[\halfpage]
	The $n-$th term in the series is of the form 
	\[ a_n = \dfrac\ve{\vc n^2 - \vd} = \dfrac{(\va n + \vb)-(\va n - \vb)}{(\va n+\vb)\cdot(\va n - \vb)} 
	= \dfrac{1}{\va n - \vb} - \dfrac{1}{\va n + \vb}  \] 
	And therefore, the sum of the first $n$ terms = $S_n$ is 
	\begin{align}
		S_n &= \sum_{k=1}^n \dfrac{1}{\va n - \vb} - 
					 \sum_{k=1}^n \dfrac{1}{\va n + \vb}  \\
			  &= \underbrace{\left( \dfrac{1}\m + \dfrac{1}\n + \dfrac{1}\o + \ldots + \dfrac{1}{\va n-\vb} \right) -
			  \left( \dfrac{1}\n + \dfrac{1}\o  + \ldots + \dfrac{1}{\va n - \vb} + \dfrac{1}{\va n+\vb} \right)}
				_{\text{middle terms cancel out}} \\
				&= 1-\dfrac{1}{2n + 1} = \dfrac{2n}{2n + 1}
	\end{align}
\end{solution}
\ifprintanswers\begin{codex}$\dfrac{2n}{2n+1}$\end{codex}\fi
