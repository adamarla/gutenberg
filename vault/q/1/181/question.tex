

\question[2] Find the inverse of the matrix $A = \left[
 \begin{array}{cc}
   a & b \\
   c & \dfrac{1+bc}{a}
 \end{array}
\right]$


\ifprintanswers
\fi 

\begin{solution}[\halfpage]
	\begin{align}
		|A| &= a\cdot\dfrac{1+bc}{a} - b\cdot c = 1 \\
		A^{-1} &= \dfrac{\text{adj}(A)}{|A|} \\
		       &= \dfrac{1}{1}\left[
		         \begin{array}{cc}
		           \dfrac{1+bc}{a} & -b \\
		           -c & a
		         \end{array}
		       \right]
	\end{align}
\end{solution}
