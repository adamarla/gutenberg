\question[1]  If $\tan A = \frac\va\vb$ then find the value of
\[ (\sin A + \cos A)\times\sec A \]

\watchout

\begin{solution}[\mcq]
	\begin{align}
		(\sin A + \cos A)\cdot\sec A &= (\sin A + \cos A)\cdot\dfrac{1}{\cos A} \\
		                             &= \dfrac{\sin A}{\cos A} + \dfrac{\cos A}{\cos A} = (\tan A + 1) \\
		                             &= \dfrac\va\vb + 1 = \dfrac\vc\vd
	\end{align}
\end{solution}
