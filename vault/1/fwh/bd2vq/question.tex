\question This question is about our Solar System. If $S$ = \{Mercury, Venus, Mars\} = 
set of planets smaller than the Earth, $G =$\{Jupiter, Saturn, Uranus, Neptune\} = set of gas planets,
$M=$\{Mars, Jupiter, Saturn, Uranus, Neptune\} = planets with > 1 moon, $N=$\{Jupiter, Saturn, Uranus, Neptune\} 
= planets with > 10 moons and $E=$\{Venus, Uranus\} = planets that rotate from east-to-west, then how 
would you express the following in set notation \textbf{and} what would they evaluate to

\textit{Pluto is now no longer considered a planet}

\insertQR{}

\begin{parts}
  \part[1] Planets that rotate from west-to-east and are gas planets?  

  \insertQR{}
\begin{solution}[\mcq]
    What we are looking for is $E'\cap G$
    \begin{align}
      &\lbrace\text{Mercury, Earth, Mars,Jupiter, Saturn, Neptune}\rbrace \nonumber\\ 
      &\cap\lbrace\text{Jupiter, Saturn, Uranus, Neptune}\rbrace=\lbrace\text{Jupiter, Saturn, Neptune}\rbrace
    \end{align}
  \end{solution}

  \part[1] Planets that have more than one moon - but are smaller than the Earth?  

  \insertQR{}
\begin{solution}[\mcq]
    $M\cap S$. 
    \begin{align}
      &\lbrace\text{Mars, Jupiter, Saturn, Uranus, Neptune}\rbrace\nonumber \\
      &\cap\lbrace\text{Mercury, Venus, Mars}\rbrace = \lbrace\text{Mars}\rbrace
    \end{align}
  \end{solution}

  \part[1] Planets that have greater than 10 moons - but are not gas planets?

  \insertQR{}
\begin{solution}[\mcq]
    $N\cap G'$ 
    \begin{align}
      &\lbrace\text{Jupiter, Saturn, Uranus, Neptune}\rbrace\nonumber \\
      &\cap\lbrace\text{Mercury, Venus, Earth, Mars}\rbrace = \phi 
    \end{align}
  \end{solution}

  \part[1] Gas planets that rotate from east to west?  

  \insertQR{}
\begin{solution}[\mcq]
    $E\cap G$
    \begin{align}
      &\lbrace\text{Venus, Uranus}\rbrace \nonumber\\ 
      &\cap\lbrace\text{Jupiter, Saturn, Uranus, Neptune}\rbrace = \lbrace\text{Uranus}\rbrace
    \end{align}
  \end{solution}

  \part[1] Planets larger than the Earth?

  \insertQR{}
\begin{solution}[\mcq]
    $S' = \mathbb{U}-S$
    \begin{align}
      =\lbrace\text{Jupiter, Saturn, Uranus, Neptune}\rbrace
    \end{align}
  \end{solution}

\end{parts}
