\ADD\vb\vc\vm
\SUBTRACT\vm\va\vn
\MULTIPLY\vb\vc\vo
\SUBTRACT\vd\vc\vp
\SUBTRACT\ve\vc\vq

\LCM\vp\vd\vf
\DIVIDE\vf\vp\vg
\DIVIDE\vf\vd\vh
\LCM\vq\ve\vi
\DIVIDE\vi\vq\vj
\DIVIDE\vi\ve\vk

\begin{solution}[\halfpage]
  If $ON = y-$intercept $=n$ and $OM = x-$intercept =$n - \vc$,
  then the equation of a line with these intercepts would be 

  \begin{align}
    \dfrac{x}{n-\vc} + \dfrac{y}{n} &= 1
  \end{align}

  Moreover, as the line passes through $A = (\va, -\vb)$,
  the following would also be true
  \begin{align}
    \dfrac{\va}{n-\vc}-\dfrac{\vb}{n} &= 1 \\
    \Rightarrow n^2 + \vn n - \vo &= 0 \\
    \Rightarrow n &= \vd,\, \ve
  \end{align}

  As there are two distinct values of $n = y-$intercept, 
  there would be two distinct lines given by 
  \begin{align}
    \frac{x}{\vp} + \frac{y}{\vd} &= 1 \Rightarrow \vg x + \vh y = \vf \\
    \frac{x}{\vq} + \frac{y}{\ve} &= 1 \Rightarrow \vj x + \vk y = \vi
  \end{align}
\end{solution}
