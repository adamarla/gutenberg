\begin{snippet}
    \correct
    
    $f(x) = \sqrt{x}$ and $g(x) = x$ change at the same rate at $x=\frac{1}{4}$ 
    \reason
    
    \begin{align}
	f'(x) = \frac{1}{2\sqrt{x}}&\text{ and } g'(x) = 1  \\
	\text{so } \frac{1}{2\sqrt{x}} = 1&\implies \sqrt{x} = \frac{1}{2} \\
	\text{or } x &= \frac{1}{4} 
\end{align}
    
\end{snippet}
