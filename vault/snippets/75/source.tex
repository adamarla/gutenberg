\begin{snippet}
    \correct
    
    \[ \sin \left(\sin^{-1}\f + \cos^{-1}\f \right) = 1 \]
    
    \reason
    
    We know that 
    \[ \qquad \sin^{-1}x + \cos^{-1} x = \frac\pi{2} \]
    
    And therefore, 
    \[ \qquad \sin^{-1} \f + \cos^{-1} \f = \frac\pi{2} \]
    Hence 
    \[ \quad \sin \left(\sin^{-1}\f + \cos^{-1}\f\right) = \sin\frac\pi{2} = 1\]
    
\end{snippet}
