\begin{snippet}
    
    \incorrect
    
    \[ f(x) = \begin{cases}
\dfrac{\tan 2x}{x},\, x\neq 0 \\ 
a,\, x = 0  
\end{cases} \]
%text

is continuous at $x=0$ if $a = 1$
%

    \reason

\[\lim_{x\to 0}\dfrac{\tan x}{x} = 1\textbf{ but } \lim_{x\to 0}\dfrac{\tan 2x}{x} = 2 \]
%text

Therefore, for $f(x)$ to be continuous
at $x=0$, $ a = \lim_{x\to 0}f(x) = 2$. \underline{Hence, $a = 2$}
%
    
\end{snippet}
