\begin{snippet}
    
    
    \incorrect
    
    \[ y = \tan^{-1} 1 + \sin^{-1}\third + \cos^{-1}\third = \frac\pi{2} \]
    
    \reason
    
    Recall the standard result 
    \[ \qquad \sin^{-1} x + \cos^{-1} x = \frac\pi{2} \]
    
    And therefore, 
    \begin{align}
	\tan^{-1} &= \frac\pi{4} \\
	\text{and } \sin^{-1} \third &+ \cos^{-1}\third = \frac\pi{2} \\
	\therefore y &= \frac\pi{4} + \frac\pi{2} = \frac{3\pi}{4} 
\end{align}
    
\end{snippet}
