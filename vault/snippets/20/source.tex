\begin{snippet}
    \correct

\[\int_0^1 \frac{dx}{1+x^2} = \left[\cot^{-1} x \right]_1^0\]
    
    \reason

\begin{align}
	\int \frac{dx}{1+x^2} = \tan^{-1} &x + C =-\cot^{-1} x + C  \\
	\therefore \int_0^1 \frac{dx}{1+x^2} &= \left[-\cot^{-1} x \right]_0^1 \\
	&= \left[\cot^{-1} x \right]_1^0 \\
	\because \int_a^b f(x)\cdot dx &= -\int_b^a f(x)\cdot dx 
\end{align}
    
\end{snippet}
