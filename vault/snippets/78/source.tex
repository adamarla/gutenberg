\begin{snippet}
    \correct
    
    \[ y = \sini{-1} + 2\tani{-\sqrt{3}} = -\frac{7\pi}{6}\]
    
    \reason
    
    \begin{center}
  \begin{tabular}{NNN}
   \toprule
        & \text{Principal Range} & \text{Value} \\
   \midrule 
   \sini{-1} & \left[-\frac\pi{2},\frac{\pi}{2} \right] & -\frac\pi{2} \\
    \midrule 
    \tani{-\sqrt{3}} & \left(-\frac\pi{2},\frac{\pi}{2} \right) & -\frac\pi{3} \\
    \bottomrule
  \end{tabular}
\end{center}
Therefore 
\[ \qquad y = -\frac\pi{2} + 2\cdot \left(-\frac\pi{3} \right) = \frac{7\pi}{6} \]
    
\end{snippet}
