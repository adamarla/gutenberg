\begin{snippet}
    \correct
    
    The rate of change of $f(x) = x^4$ at $x = 2$ is eight times faster than at $x=1$ 
    
    \reason
    
    \begin{align}
	f(x) = x^4 &\implies f'(x) = 4x^3 \\
	\therefore \frac{f'(2)}{f'(1)} &= \frac{4\cdot 2^3}{4\cdot 1^3} = 8 
\end{align}
    
\end{snippet}
