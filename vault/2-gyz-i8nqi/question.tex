\question[6] If $x+y+z=0$, verify that $x^3+y^3+z^3=3xyz$.
\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
  \marginnote[3cm] {If you remember the expansion for $x+y+z)^3$, that's great. If not, you'll need to do some furious scribbling in the rough work column to get \textit{Step(1)}. Ultimately though, however you expand it, the equation is symmetrical about \textit{x, y and z}. So you want to get it in a form that exploits this symmetry.}
\fi 
\begin{solution}[\fullpage]
  \begin{align}  
	  (x+y+z)^3 = x^3&+y^3+z^3 \nonumber \\
	  			 	 &+3xy(x+y)+3yz(y+z)+3zx(z+x) \nonumber \\
	  			 	 &+6xyz \\
			  0 = x^3&+y^3+z^3 \nonumber \\
	  				 &+3xy(x+y)+2xyz \nonumber \\
	  				 &+3yz(y+z)+2xyz \nonumber \\
	  				 &+3zx(z+x)+2xyz
  \end{align}  
  Now add $3xyz$ to both sides of the equation, to get it into the form expected in the final result,
  \begin{align}
	  3xyz = x^3&+y^3+z^3 \nonumber \\
	  			&+3xy(x+y)+3xyz \nonumber \\
	  			&+3yz(y+z)+3xyz \nonumber \\
	  			&+3zx(z+x)+3xyz \\
	  3xyz = x^3&+y^3+z^3 \nonumber \\
	  			&+3xy(x+y+z) \nonumber \\
	  			&+3yz(x+y+z) \nonumber \\
	  			&+3zx(x+y+z) \\
	  3xyz = x^3&+y^3+z^3
  \end{align}
\end{solution}
