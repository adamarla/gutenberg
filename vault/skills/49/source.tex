\textcolor{blue}{Standard Integrals \#2}

Integrals of inverse trigonometric functions 
\end{skill}

%

\newcard

%text
The following integrals follow straight
the derivatives of inverse trigonometric
functions 

%
\begin{center}
\begin{tabular}{MM} 
\toprule
\text{Derivative} & \text{Integral} \\
\midrule 
\frac{d}{dx}\sin^{-1}x = \frac{1}{\sqrt{1-x^2}} & \int \frac{dx}{\sqrt{1-x^2}} = \sin^{-1} x + C \\
\midrule 
\frac{d}{dx}\cos^{-1} x = -\frac{1}{\sqrt{1-x^2}} & \int\frac{dx}{\sqrt{1-x^2}} = -\cos^{-1} x + C \\
\midrule 
\frac{d}{dx}\tan^{-1}x = \frac{1}{1+x^2} & \int\frac{dx}{1+x^2} = \tan^{-1} x + C \\
\midrule 
\frac{d}{dx}\cot^{-1} x = \frac{-1}{1+x^2} & \int\frac{dx}{1+x^2} = -\cot^{-1} x + C \\
\midrule 
\frac{d}{dx}\sec^{-1} x = \frac{1}{x\sqrt{x^2-1}} & \int\frac{dx}{x\sqrt{x^2 - 1}} = \sec^{-1} x + C \\
\midrule 
\frac{d}{dx}\csc^{-1} x = \frac{-1}{x\sqrt{x^2-1}} & \int\frac{dx}{x\sqrt{x^2 - 1}} = -\csc^{-1} x + C  \\
\bottomrule
\end{tabular}
\end{center}
