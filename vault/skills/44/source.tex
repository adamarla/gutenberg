\begin{align}
\tan^{-1}x + \tan^{-1}y &= \tan^{-1}\left( \dfrac{x+y}{1-xy}\right) \\
\tan^{-1}x - \tan^{-1}y &= \tan^{-1}\left( \dfrac{x-y}{1+xy}\right) \\
2\tan^{-1} x &= \sin^{-1}\dfrac{2x}{1+x^2} \\
2\tan^{-1} x &= \cos^{-1}\dfrac{1-x^2}{1+x^2} \\
2\tan^{-1} x &= \tan^{-1}\dfrac{2x}{1-x^2} \\
\end{align}
