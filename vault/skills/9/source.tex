\begin{skill}
\begin{narrow}
\[ \lim_{x\to a}\dfrac{x^n - a^n}{x-a} = ? \]
\end{narrow}

\reason 

\[ \qquad \lim_{x\to a}\dfrac{x^n-a^n}{x-a} = n\cdot a^{n-1} \]
%text

The above results holds under the following
two conditions \newline 

%
\begin{center}
\begin{tabular}{MMM}
\toprule
 n & a & \text{Example}  \\
 \midrule 
 n\in\mathbb{Z}^+ & \text{Anything} & \lim_{x\to a}\dfrac{x^5-a^5}{x-a} = 5\cdot a^4\\
 \midrule 
 n\in\mathbb{Q}& a > 0&  \lim_{x\to a}\dfrac{x^{\frac{1}{3}}-a^{\frac{1}{3}}}{x-a} = \frac{1}{3}\cdot a^{-\frac{2}{3}}\\
 \bottomrule
\end{tabular} 
\end{center}
%text

where $\mathbb{Z}^+$ is the set of \underline{positive integers}
and $\mathbb{Q}$ is the set of \underline{rational numbers} 
(things like $\frac{3}{4}, -\frac{7}{5}$ etc.)
%
\end{skill}
