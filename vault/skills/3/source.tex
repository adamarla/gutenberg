\begin{skill}
\begin{narrow}
\textcolor{blue}{Mathematical symbols \& notations}

Ability to \underline{read \& understand} things like the following 
\[
	\quad f(x) = \sum_{k=1}^N \dfrac{1}{x^k},\,x\in\mathbb{R}, N\in\mathbb{Z}^+
\]
\end{narrow} 

\reason 

%text
Mathematical symbols allow you to express 
complex ideas \underline{quickly, compactly and precisely}\newline

Hence, it would help you to be familiar with 
at least the basic symbols below. \newline

\begin{tabular}{McM}
\midrule 
\text{Symbol} & Meaning & \text{Example} \\ 
\midrule 
\therefore & Therefore & \therefore a = b \\ 
\midrule 
\because & Because & \frac{1}{x} < \frac{1}{y}\because x > y \\
\midrule 
\forall & For all & \forall\,  x\in [2,9) \\ 
\midrule 
\mathbb{R} & Real numbers & \forall\, x\in\mathbb{R} \\
\midrule 
\mathbb{Z} & Integers & \forall\, z\in\mathbb{Z} \\
\midrule 
\sum & Sum & \frac{1}{x} + \frac{2}{x^2} +\ldots = \sum_{k=1}^\infty\frac{k}{x^k} \\
\midrule 
\in & Belongs to & x\in\mathbb{Z}^+ \\
\midrule
\implies & Implies & \frac{1}{x} > 0\implies x > 0 \\
\midrule
 \end{tabular} 
\end{skill}
